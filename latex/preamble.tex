% ===========================================
% Section heading formatting
% ===========================================
\usepackage{titlesec}

% Chapter: "1. Title" on one line
\titleformat{\chapter}[hang]
  {\normalfont\LARGE\bfseries}
  {\thechapter.}
  {0.5em}
  {}
\titlespacing*{\chapter}{0pt}{-10pt}{10pt}

% Section
\titleformat{\section}
  {\normalfont\Large\bfseries}
  {\thesection.}
  {0.5em}
  {}
\titlespacing*{\section}{0pt}{8pt}{4pt}

% Subsection
\titleformat{\subsection}
  {\normalfont\large\bfseries}
  {\thesubsection.}
  {0.5em}
  {}
\titlespacing*{\subsection}{0pt}{6pt}{2pt}

% Subsubsection
\titleformat{\subsubsection}
  {\normalfont\normalsize\bfseries}
  {\thesubsubsection.}
  {0.5em}
  {}
\titlespacing*{\subsubsection}{0pt}{4pt}{2pt}

% ===========================================
% Force figures to stay in place (no floating)
% ===========================================
\usepackage{float}
\floatplacement{figure}{H}    % Here, definitely
\floatplacement{table}{H}     % Here, definitely

% Redefine figure environment to not float
\let\origfigure\figure
\let\endorigfigure\endfigure
\renewenvironment{figure}[1][H]{
  \origfigure[H]
}{
  \endorigfigure
}

% Same for tables
\let\origtable\table
\let\endorigtable\endtable
\renewenvironment{table}[1][H]{
  \origtable[H]
}{
  \endorigtable
}
