% ===========================================
% Remove blank pages
% ===========================================
\usepackage{etoolbox}
\makeatletter
\patchcmd{\chapter}{\if@openright\cleardoublepage\else\clearpage\fi}{\clearpage}{}{}
\makeatother
\let\cleardoublepage\clearpage

% ===========================================
% Chapter heading formatting
% ===========================================
\usepackage{titlesec}

% Remove "Chapter X" label, use "X. Title" on one line
\titleformat{\chapter}[block]
  {\normalfont\LARGE\bfseries}
  {\thechapter.\quad}
  {0pt}
  {}
\titlespacing*{\chapter}{0pt}{0pt}{20pt}

% Remove uppercase from headers
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\nouppercase{\leftmark}}
\fancyhead[R]{\thepage}
\renewcommand{\headrulewidth}{0.4pt}

% Redefine chaptermark to not uppercase
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}

% ===========================================
% Unnumbered sections in frontmatter
% ===========================================
% Make sure frontmatter sections are not numbered
\let\oldfrontmatter\frontmatter
\renewcommand{\frontmatter}{%
  \oldfrontmatter
  \renewcommand{\thesection}{}%
}
